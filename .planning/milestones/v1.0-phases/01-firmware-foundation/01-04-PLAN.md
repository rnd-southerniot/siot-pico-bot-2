```python
from machine import WDT
import uasyncio
import utime

class WatchdogKeeper:
    """
    Two-layer safety system:
    1. Hardware WDT (machine.WDT): Resets entire device if feed_loop stops running.
       Catches full event loop hangs (runaway student code that blocks indefinitely).
    2. Software motor timeout: Stops motors after MOTOR_TIMEOUT_S seconds of running.
       Catches student programs that run too long without explicitly stopping.

    IMPORTANT: Create this AFTER IMU calibration completes (Pitfall 1 from research).
    """

    MOTOR_TIMEOUT_S = 30   # Stop motors after 30s — configurable per-unit

    def __init__(self, timeout_ms: int = 8000):
        # WDT timeout: 1000-8388ms on RP2040. 8000ms gives 4s margin with 4s feed interval.
        self._wdt = WDT(timeout=timeout_ms)
        self._wdt.feed()  # Feed immediately after creation
        self._motor_start_time = None
        self._motors_running = False

    async def feed_loop(self):
        """
        Feed WDT every 4 seconds. This coroutine MUST keep running.
        If the event loop hangs (student code blocks cooperative scheduling),
        this coroutine stops executing and the WDT fires after 8 seconds,
        resetting the device.
        """
        while True:
            self._wdt.feed()
            await uasyncio.sleep_ms(4000)

    def arm_motor_timeout(self):
        """Call when student program starts motor movement."""
        self._motor_start_time = utime.time()
        self._motors_running = True

    def disarm_motor_timeout(self):
        """Call when student program explicitly stops motors."""
        self._motor_start_time = None
        self._motors_running = False

    def check_motor_timeout(self, stop_fn) -> bool:
        """
        Call from motor_pid_loop on each iteration.
        If motors have run longer than MOTOR_TIMEOUT_S, calls stop_fn() and disarms.
        Returns True if timeout triggered (motors were stopped).
        stop_fn: callable that brakes both motors, e.g. lambda: (left.brake(), right.brake())
        """
        if not self._motors_running or self._motor_start_time is None:
            return False
        elapsed = utime.time() - self._motor_start_time
        if elapsed > self.MOTOR_TIMEOUT_S:
            stop_fn()
            self.disarm_motor_timeout()
            print(f"[SAFETY] Motor timeout after {elapsed}s — motors stopped")
            return True
        return False

    def emergency_stop(self, stop_fn):
        """
        Immediate motor stop. Call from exception handlers anywhere in firmware.
        Disarms timeout so timeout doesn't double-fire.
        """
        stop_fn()
        self.disarm_motor_timeout()
        print("[SAFETY] Emergency stop triggered")
```
