```python
from lib.mpu6050 import MPU6050
from machine import I2C, Pin
import config

class IMUHAL:
    def __init__(self, i2c: I2C):
        self._imu = MPU6050(i2c)

    def calibrate(self, samples=200):
        """Call at boot BEFORE WDT starts."""
        self._imu.calibrate_gyro_z(samples=samples)

    def gyro_z_dps(self) -> float:
        """Gyro Z rate in degrees/second (calibrated)."""
        return self._imu.gyro_z_calibrated()

    def accel(self) -> tuple:
        return self._imu.accel()
```

HeadingTracker class (integrates gyro Z into heading angle):
```python
import uasyncio
import utime

class HeadingTracker:
    def __init__(self, imu_hal: IMUHAL):
        self._imu = imu_hal
        self._heading = 0.0
        self._last_t = utime.ticks_ms()

    async def update_loop(self):
        """100Hz heading integration. Run as async task."""
        while True:
            now = utime.ticks_ms()
            dt = utime.ticks_diff(now, self._last_t) / 1000.0
            self._last_t = now
            if dt > 0:
                self._heading += self._imu.gyro_z_dps() * dt
            await uasyncio.sleep_ms(10)  # 100Hz

    def get_heading(self) -> float:
        return self._heading

    def reset(self):
        self._heading = 0.0
```

IRLineSensor (FW-04):
```python
import uasyncio
from machine import Pin, ADC

class IRLineSensor:
    def __init__(self, pins: list):
        """pins: list of GPIO pin numbers (ADC-capable: GP26, GP27, GP28)"""
        self._adcs = [ADC(Pin(p)) for p in pins]

    async def read_all(self) -> list:
        """Returns list of 0-65535 per channel. Non-blocking."""
        await uasyncio.sleep_ms(0)  # yield once before reading
        return [adc.read_u16() for adc in self._adcs]
```

UltrasonicSensor (FW-05):
```python
import uasyncio
import utime
from machine import Pin

class UltrasonicSensor:
    TIMEOUT_US = 30000  # ~5m max range

    def __init__(self, trig_pin: int, echo_pin: int):
        self._trig = Pin(trig_pin, Pin.OUT)
        self._echo = Pin(echo_pin, Pin.IN)
        self._trig.value(0)

    async def read_cm(self) -> float:
        """Returns distance in cm. Returns -1.0 on timeout. Non-blocking."""
        self._trig.value(0)
        await uasyncio.sleep_ms(2)      # settling
        self._trig.value(1)
        await uasyncio.sleep_ms(0)      # 10us high pulse (MicroPython yields; actual pulse ~10-50us)
        self._trig.value(0)

        start = utime.ticks_us()
        while self._echo.value() == 0:
            await uasyncio.sleep_ms(0)
            if utime.ticks_diff(utime.ticks_us(), start) > self.TIMEOUT_US:
                return -1.0

        pulse_start = utime.ticks_us()
        while self._echo.value() == 1:
            await uasyncio.sleep_ms(0)
            if utime.ticks_diff(utime.ticks_us(), pulse_start) > self.TIMEOUT_US:
                return -1.0

        duration_us = utime.ticks_diff(utime.ticks_us(), pulse_start)
        return duration_us / 58.0
```

ColorSensor (FW-06) — support both I2C (TCS34725) and analog fallback:
```python
class ColorSensor:
    def __init__(self, i2c=None, analog_pin: int = None):
        """Use i2c for TCS34725, or analog_pin for simple light sensor.
        Only one mode active at a time."""
        self._i2c = i2c
        self._analog = ADC(Pin(analog_pin)) if analog_pin is not None else None

    async def read(self) -> dict:
        """Returns {'r':0,'g':0,'b':0,'c':0} for I2C or {'lux':0} for analog."""
        await uasyncio.sleep_ms(0)
        if self._analog:
            return {"lux": self._analog.read_u16()}
        if self._i2c:
            # TCS34725 register reads (0x94=RDATAL, 0x96=GDATAL, 0x98=BDATAL, 0x94=CDATAL)
            # Read 2 bytes each at I2C addr 0x29
            try:
                def rd(reg):
                    self._i2c.writeto(0x29, bytes([0x80 | reg]))
                    d = self._i2c.readfrom(0x29, 2)
                    return d[0] | (d[1] << 8)
                return {"c": rd(0x14), "r": rd(0x16), "g": rd(0x18), "b": rd(0x1A)}
            except Exception:
                return {"r": 0, "g": 0, "b": 0, "c": 0}
        return {"lux": 0}
```
