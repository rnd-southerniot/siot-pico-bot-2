Use the standard RP2040 PIO quadrature counter program. This is the CORRECT working program — not the illustrative sketch in RESEARCH.md (the research explicitly says "use the community example verbatim"):

```python
import rp2
from machine import Pin
import utime

@rp2.asm_pio()
def quadrature_program():
    # Standard RP2040 quadrature encoder PIO program
    # State: X = previous AB, ISR accumulates count
    # Each state transition: increment or decrement counter
    # Full implementation using jmp table for direction
    set(x, 0)        # init counter to 0
    wrap_target()
    # Read both encoder pins
    in_(pins, 2)     # read A and B into ISR (2 bits)
    mov(y, isr)      # save new state in Y
    jmp(x_not_y, "changed")
    jmp("wrap_target")
    label("changed")
    # Push count on change so Python can drain FIFO
    push(noblock)    # push ISR to RX FIFO
    mov(x, y)        # update previous state
    wrap()
```

IMPORTANT: The above PIO assembly is illustrative. Before implementing, use the actual working RP2040 PIO quadrature program from the MicroPython examples repository. The key behavioral requirements are:
1. Reads A and B encoder pins simultaneously
2. Detects direction from state transitions (CW increments, CCW decrements)
3. Counts in hardware — Python only reads accumulated count from FIFO
4. Uses rp2.StateMachine with in_base=Pin(pin_a) so both A and B are read sequentially

The EncoderPIO class:
```python
class EncoderPIO:
    def __init__(self, pin_a: int, pin_b: int, sm_id: int = 0):
        self._sm = rp2.StateMachine(sm_id, quadrature_program, in_base=Pin(pin_a), freq=1_000_000)
        self._sm.active(1)
        self._count = 0
        self._last_count = 0
        self._last_time = utime.ticks_ms()

    def count(self) -> int:
        while self._sm.rx_fifo():
            v = self._sm.get()
            # Interpret as signed 32-bit (PIO uses 2s complement for decrements)
            if v > 0x7FFFFFFF:
                v -= 0x100000000
            self._count += v
        return self._count

    def reset(self):
        self._count = 0
        self._last_count = 0

    def delta(self) -> int:
        """Ticks since last delta() call."""
        current = self.count()
        d = current - self._last_count
        self._last_count = current
        return d

    def rpm(self, dt: float) -> float:
        """Revolutions per minute given actual elapsed time dt in seconds."""
        if dt <= 0:
            return 0.0
        import config
        ticks = self.delta()
        revs_per_sec = ticks / config.TICKS_PER_REV_QUAD / dt
        return revs_per_sec * 60.0

    def deinit(self):
        self._sm.active(0)
```

Using PIO block 1 avoids conflict. If sm_id 4-7 are unavailable in the target MicroPython build, fall back to sm_id=2 and sm_id=3 with a comment explaining the conflict risk.

Create hal/motors.py — thin wrapper around lib/motor.py that applies K-8 speed cap:

```python
from lib.motor import Motor
import config

# K-8 safety: cap applied at HAL boundary — cannot be bypassed by higher layers
# Default 70% (config.MOTOR_MAX_SPEED_PCT). Make configurable via config.py.
_MAX_SPEED = getattr(config, 'MOTOR_MAX_SPEED_PCT', 70) / 100.0

class MotorHAL:
    def __init__(self, pin_a: int, pin_b: int):
        self._motor = Motor(pin_a, pin_b)

    def drive(self, speed: float):
        """Drive motor. Speed -1.0 to 1.0. Clamped to ±MOTOR_MAX_SPEED_PCT."""
        clamped = max(-_MAX_SPEED, min(_MAX_SPEED, speed))
        self._motor.drive(clamped)

    def brake(self):
        self._motor.brake()

    def coast(self):
        self._motor.coast()

    def deinit(self):
        self._motor.deinit()
```

Add MOTOR_MAX_SPEED_PCT = 70 to config.py if not already present (check first; add only if missing).
